/**
* 思路：
       数组长度为1：直接返回这个元素
       否则：先快排，然后判断中间元素与它的后一个元素是否相等，相等说明这个数存在，否则不存在。
           
*/
public class Solution {
    public int MoreThanHalfNum_Solution(int [] array) {
        quickSort(array,0,array.length-1);
        int middleIndex = (array.length - 1) / 2;
        if(array.length == 1){
            return array[0];
        }else {
            if(array[middleIndex] == array[middleIndex+1]){
                return array[middleIndex];
            }else {
                return 0;
            }
        }
    }
    
    public  void quickSort(int[] array,int low,int high){
        if(low < high){
            int posiontion = oneFastSort(array,low,high);
            quickSort(array,low,posiontion-1);
            quickSort(array,posiontion+1,high);
        }
    }
    public  int oneFastSort(int[] array,int low,int high){
        int key,flag=0;
        key = array[low];
        while(low < high){
            if(flag==0){
                if(array[high] > key){
                    high--;
                }else {
                    array[low++] = array[high];
                    flag = 1;
                }
            }else {
                if(array[low] < key){
                    low++;
                }else {
                    array[high--] = array[low];
                    flag = 0;
                }
            }
        }
        array[low] = key;
        return low;
    }
    
}
