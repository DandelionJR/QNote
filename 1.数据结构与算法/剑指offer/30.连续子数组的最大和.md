    1.暴力枚举版：直接两层循环，遍历所有情况并记录最大值。
    2.优化版：
        temp表示和为正数的最长的连续子数组的和（如果该连续子数组和为负数，则开始新的序列；
        如果和为正数，则继续拼上新的元素）
        greatestSum表示连续子数组的最大和（每遍历一个元素，我们就取temp和greatestSum的最大值）
*/
public class Solution {
    //优化版
    public int FindGreatestSumOfSubArray(int[] array) {
        int temp = array[0];
        int greatestSum = array[0];
        for (int i = 1; i < array.length; i++) {
            temp = temp < 0 ? array[i] : temp+array[i];
            greatestSum = greatestSum > temp ? greatestSum : temp;
        }
        return greatestSum;
    }
