```java
/*
    思路：
        当树不为空时，交换其左孩子和右孩子，并对他们的左孩子和右孩子也做同样的处理。
*/
/**
public class TreeNode {
    int val = 0;
    TreeNode left = null;
    TreeNode right = null;

    public TreeNode(int val) {
        this.val = val;

    }

}
*/
public class Solution {
    public void Mirror(TreeNode root) {
        TreeNode temp;
       if(root != null){
            temp = root.left;
            root.left = root.right;
            root.right = temp; 
            Mirror(root.left);
            Mirror(root.right);
       }
    }
}
```
